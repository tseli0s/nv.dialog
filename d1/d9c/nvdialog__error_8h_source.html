33/**
34 * @brief An enumerator describing possible errors that may occur from the library.
35 * One of its fields is returned by `nvd_get_error` depending on whether a (and what)
36 * problem might've occured after an operation.
37 * @sa nvd_stringify_error
38 * @since v0.1.0
39 */
40typedef enum {
41 NVD_NO_ERROR = 0,
42 NVD_NO_DISPLAY,
43 NVD_BACKEND_FAILURE,
44 NVD_INVALID_PARAM,
45 NVD_NOT_INITIALIZED,
46 NVD_BACKEND_INVALID,
47 NVD_FILE_INACCESSIBLE,
48 NVD_STRING_EMPTY,
49 NVD_OUT_OF_MEMORY,
50 NVD_INTERNAL_ERROR,
51 NVD_ALREADY_INITIALIZED,
52} NvdError;
53
54/**
55 * @brief Returns the current error code of the library.
56 *
57 * The error code has to be manually be transformed into a string
58 * using @ref nvd_stringify_error. Else, use the @ref NvdError enum to check for
59 * the error manually.
60 *
61 * @ingroup Error
62 * @returns The error code (Will be automatically 0 if no errors were
63 * encountered).
64 */
65NVD_API NvdError nvd_get_error(void);
66
67/**
68 * @brief Transforms an error code into a string representation, that can be
69 * used to print errors to the console.
70 * @returns A @ref NvdDynamicString containing the error as a string, or NULL in case of failure or invalid parameter.
71 * @ingroup Error
72 * @note Pre v0.10 code may not work with this function due to different return types.
73 */
74NVD_API NvdDynamicString *nvd_stringify_error(NvdError err);
