28/**
29 * @brief An opaque file dialog type, representing either a file
30 * dialog that can be used to open a file or save a file, and since v0.9.0,
31 * opening a directory instead.
32 * @since v0.1.0
33 * @ingroup FileDialog
34 */
35typedef struct _NvdFileDialog NvdFileDialog;
36
37/**
38 * @brief Creates a new, empty @ref NvdFileDialog to be used for
39 * retrieving a file and returns it.
40 *
41 * This creates a new file dialog and returns it. In order to use it, you must
42 * call
43 * @ref nvd_get_file_location and pass a pointer to write the file location to.
44 * @note This function creates problems with the Adwaita backend. Those problems
45 * should be fixed by the time v0.6.0 is officially released.
46 *
47 * @param title A string to put as the dialog title.
48 * @param file_extensions Unused legacy parameter, just pass NULL in here.
49 * @returns An empty @ref NvdFileDialog object if successful, otherwise NULL and
50 * an error retrievable through @ref nvd_get_error is set.
51 * @ingroup FileDialog
52 */
53NVD_API NvdFileDialog *nvd_open_file_dialog_new(const char *title,
54 const char *file_extensions);
55
56/**
57 * @brief Creates a new, empty @ref NvdFileDialog object that will be used to
58 * save a file in the selected (from the user) location.
59 * @param title The title of the dialog, default is "Save file"
60 * @param default_filename The default filename to use for saving.
61 * @return An @ref NvdFileDialog on success, otherwise NULL and @ref
62 * nvd_get_error should be called to get the failure reason.
63 * @ingroup FileDialog
64 */
65NVD_API NvdFileDialog *nvd_save_file_dialog_new(const char *title,
66 const char *default_filename);
67
68/**
69 * @brief Creates a new, empty @ref NvdFileDialog object that will be used to
70 * request a directory or folder in the filesystem.
71 *
72 * @param title The title of the dialog, default is "Open directory"
73 * @param default_folder Path to the default folder to open. May be ignored in
74 * some platforms.
75 * @return An @ref NvdFileDialog on success or NULL otherwise. @ref
76 * nvd_get_error should be called for further information about the error
77 * occurring.
78 * @ingroup FileDialog
79 * @since v0.9.0
80 */
81NVD_API NvdFileDialog *nvd_open_folder_dialog_new(const char *title,
82 const char *default_folder);
83
84/**
85 * @brief Returns the filesystem path chosen through the @ref NvdFileDialog
86 * passed.
87 *
88 * @details This function will return the path on the filesystem from the dialog chosen,
89 * that you can then use to either open or save the file given. It works with
90 * both save and open file dialog types.
91 *
92 * @sa nvd_open_file_dialog_new
93 * @param dialog The file dialog to take the filename from.
94 * @returns A @ref NvdDynamicString if a file/folder was selected, or NULL if no path was selected by the user.
95 * @ingroup FileDialog
96 */
97NVD_API NvdDynamicString* nvd_get_file_location(NvdFileDialog *dialog);
98
99/**
100 * @brief Returns the raw object behind the dialog.
101 * @param dialog The dialog to retrieve the object from.
102 * @return void* The raw toolkit-created object.
103 * @ingroup FileDialog
104 */
105NVD_API void *nvd_open_file_dialog_get_raw(NvdFileDialog *dialog);
