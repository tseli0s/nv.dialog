32/**
33 * @brief Possible types of NvDialog notifications. Each field will create a
34 * slightly different dialog matching the requested type.
35 * @note Some systems may not provide compatible icons "out-of-the-box", usually minimalist Linux distros.
36 * @ingroup Notification
37 * @since v0.5.0
38 * @sa NvdNotification
39 */
40typedef enum {
41 NVD_NOTIFICATION_SIMPLE, /**< A simple notification eg. to notify of
42 something new in the app. */
43 NVD_NOTIFICATION_WARNING, /**< A warning notification. */
44 NVD_NOTIFICATION_ERROR /**< An error notification, preferably used
45 before crashing. */
46} NvdNotifyType;
47
48/**
49 * @brief The base notification type used by NvDialog.
50 *
51 * NvDialog offers since v0.5.0 support for basic notifications that
52 * add an extra cross-platform solution. The notification API is cross-platform
53 * and should reduce the required setup to send a simple notification. On GNU/Linux,
54 * `libnotify` is OPTIONALLY used at runtime to send notifications, meaning it's not a
55 * hard dependency. In addition, it can be disabled from being loaded entirely at initialization
56 * by setting the environment variable NVD_NO_NOTIFS to any value (except 0). Note that if you
57 * need to create a notification down the line it will load instances of the library separately,
58 * but only if you insist so.
59 *
60 * @ingroup Notification
61 * @sa nvd_notification_new
62 * @since v0.5.0
63 */
64typedef struct _NvdNotification NvdNotification;
65
66/**
67 * @brief Creates a new notification object and returns it.
68 * @param title The title of the notification, can be NULL if no title is
69 desired.
70 * @param msg The message to show (Required, can't be NULL).
71 * @param type The type of the notification, see @ref NvdNotifyType.
72 * <b>Example:</b>
73 * @code
74 #include <nvdialog/nvdialog.h>
75 int main(int argc, char** argv) {
76 nvd_init(argv[0]);
77 NvdNotification *notif = nvd_notification_new("Notification",
78 "A notification for your
79 app.", NVD_NOTIFICATION_SIMPLE); if (!notif) return -1;
80 nvd_send_notification(notif);
81 nvd_delete_notification(notif);
82 return 0;
83 }
84 * @endcode
85 * @ingroup Notification
86 * @return An empty notification object.
87 */
88NVD_API NvdNotification* nvd_notification_new(const char* title,
89 const char* msg,
90 NvdNotifyType type);
91
92/**
93 * @brief Sends the notification to the system.
94 * It's safe to assume that the same notification can be sent multiple times.
95 * Just make sure the notification is still valid (Not freed yet).
96 *
97 * @param notification The notification object to use.
98 * @ingroup Notification
99 * @since v0.5.0
100 */
101NVD_API void nvd_send_notification(NvdNotification* notification);
102
103/**
104 * @brief Deletes a notification object from NvDialog.
105 * @note You should only use this to free notifications. <b>Do not use @ref
106 * nvd_free_object please.</b>
107 * @param notification The notification object to delete.
108 * @ingroup Notification
109 */
110NVD_API void nvd_delete_notification(NvdNotification* notification);
111
112/**
113 * @brief Adds another button with a specified action to do when it is clicked.
114 * @note Please make sure you don't register the same action twice on the same
115 * notification.
116 * @param notification The notification to add the said action to.
117 * @param action A string defining the action as well as the label of the button
118 * for the action.
119 * @param value_to_set The value to set when the action is triggered.
120 * @param value_to_return A pointer to an integer to save the value passed to
121 * @ref value_to_set .
122 * @ingroup Notification
123 */
124NVD_API void nvd_add_notification_action(NvdNotification* notification,
125 const char* action, int value_to_set,
126 int* value_to_return);
