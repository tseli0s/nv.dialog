35/**
36 * @brief Possible types of NvDialog notifications. Each field will create a slightly
37 * different dialog matching the requested type.
38 * @since v0.5.0
39 * @sa NvdNotification
40 */
41typedef enum {
42 NVD_NOTIFICATION_SIMPLE, /**< A simple notification eg. to notify of something new in the app. */
43 NVD_NOTIFICATION_WARNING, /**< A warning notification. */
44 NVD_NOTIFICATION_ERROR /**< An error notification, preferably used before crashing. */
45} NvdNotifyType;
46
47/**
48 * @brief The base notification type used by NvDialog.
49 *
50 * NvDialog offers since v0.5.0 support for basic notifications that
51 * will be useful for anything.
52 * @sa nvd_notification_new
53 * @since v0.5.0
54 */
55typedef struct _NvdNotification NvdNotification;
56
57/**
58 * @brief Creates a new notification object and returns it.
59 * @param title The title of the notification, can be NULL if no title is desired.
60 * @param msg The message to show (Required, can't be NULL).
61 * @param type The type of the notification, see @ref NvdNotifyType.
62 * <b>Example:</b>
63 * @code
64 #include <nvdialog/nvdialog.h>
65 int main(int argc, char** argv) {
66 nvd_init(argv[0]);
67 NvdNotification *notif = nvd_notification_new("Notification",
68 "A notification for your app.",
69 NVD_NOTIFICATION_SIMPLE);
70 if (!notif) return -1;
71 nvd_send_notification(notif);
72 nvd_delete_notification(notif);
73 return 0;
74 }
75 * @endcode
76 * @return An empty notification object.
77 */
78NVD_API NvdNotification* nvd_notification_new(const char *title,
79 const char *msg,
80 NvdNotifyType type);
81
82/**
83 * @brief Sends the notification to the system.
84 * @param notification The notification object to use.
85 * @since v0.5.0
86 */
87NVD_API void nvd_send_notification(NvdNotification* notification);
88
89/**
90 * @brief Deletes a notification object from NvDialog.
91 * @note You should only use this to free notifications. <b>Do not use @ref nvd_free_object please.</b>
92 * @param notification The notification object to delete.
93 */
94NVD_API void nvd_delete_notification(NvdNotification* notification);

96/**
97 * @brief Adds another button with a specified action to do when it is clicked.
98 * @note Please make sure you don't register the same action twice on the same notification.
99 * @param notification The notification to add the said action to.
100 * @param action A string defining the action as well as the label of the button for the action.
101 * @param value_to_set The value to set when the action is triggered.
102 * @param value_to_return A pointer to an integer to save the value passed to @ref value_to_set .
103 */
104NVD_API void nvd_add_notification_action(NvdNotification* notification,
105 const char* action,
106 int value_to_set,
107 int* value_to_return);
